void setup(){
	Serial.begin(9600);
	pca1.begin();
	pca2.begin();
	pca3.begin();
	pca1.setPWMFreq(60);
	pca2.setPWMFreq(60);
	pca3.setPWMFreq(60);
}

void loop(){
	//inicializar los servos
	for(int i=0;i<=15;i++){
		pca1.setPWM(i,0,90);
		pca2.setPWM(i,0,90);
		pca3.setPWM(i,0,90);
	}
	//notas
	//Primer compás
	tocarNota(pca1,D3,500);
	tocarNota(pca1,F3,500);
	tocarNota(pca1,D4,2000);

	//Segundo compás
	tocarNota(pca1,D3,500);
	tocarNota(pca1,F3,500);
	tocarNota(pca1,D4,2000);

	//Tercer compás
	tocarNota(pca1,E4,1000);
	tocarNota(pca1,F4,500);
	tocarNota(pca1,E4,500);
	tocarNota(pca1,E4,500);

	//Cuarto compás
	tocarNota(pca1,E4,500);
	tocarNota(pca1,C3,500);
	tocarNota(pca1,A3,2000);
}

//Método para tocar una nota, se le indica qué placa es, el puerto del servomotor y el delay
void tocarNota(Adafruit_PWMServoDriver placa,int nota, int duracion){
	placa.setPWM(nota,0,140);
	delay(duracion);
	pca1.setPWM(nota,0,90);
}
